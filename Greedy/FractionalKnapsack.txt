Fractional-Knapsack (w[1..n], p[1..n], W)   
    for i = 1 to n   
    do x[i] = 0   
    weight = 0   
    for i = 1 to n   
    if weight + w[i] > W then    
        x[i] = 1   
        weight = weight + w[i]   
    else   
        x[i] = (W - weight) / w[i]   
        weight = W   
        break   
    return x
